To ensure the droppable only becomes static when colliding with a surface, we'll add a ground check using a RayCast3D. Here's the modified code:

gdscript
extends RigidBody3D
class_name DroppableBase

@export var stillness_threshold: int = 10
@export var time_until_static: float = 4.0
@export var ground_check_ray_length: float = 0.4

var time_since_moved: float = 0.0
var is_static: bool = false
var ground_ray: RayCast3D

signal spawn_new_enabled

func _ready():
    # Set up ground detection raycast
    ground_ray = RayCast3D.new()
    add_child(ground_ray)
    ground_ray.target_position = Vector3(0, -ground_check_ray_length, 0)
    ground_ray.enabled = true

func _physics_process(delta: float) -> void:
    if is_static:
        return

    var is_moving: bool = linear_velocity.length() > stillness_threshold

    if is_moving:
        time_since_moved = 0.0
    else:
        time_since_moved += delta
        # Only make static if we're grounded and still for long enough
        if time_since_moved >= time_until_static and ground_ray.is_colliding():
            make_static()

    # Check if droppable is off the platform
    if global_transform.origin.y < -10:
        enable_new_spawn()

func make_static():
    is_static = true
    var static_body = StaticBody3D.new()

    get_parent().add_child(static_body)
    static_body.global_transform = global_transform

    for child in get_children():
        if child is CollisionShape3D or child is CollisionPolygon3D or child is Node3D:
            var duplicated_child: Node = child.duplicate()
            static_body.add_child(duplicated_child)
            duplicated_child.transform = child.transform
            duplicated_child.owner = static_body
            child.queue_free()

    print("Droppable is now static")
    enable_new_spawn()

func enable_new_spawn() -> void:
    spawn_new_enabled.emit()
    queue_free()

Key changes:

  1. Added a RayCast3D that points downward
  2. Check raycast collision before making static
  3. The object will only become static if:
    • It's been still for time_until_static seconds
    • The raycast detects ground below
    • Velocity is below threshold

The raycast length can be adjusted using the ground_check_ray_length property in the inspector. Make sure your object's origin is positioned at its base for accurate ground detection.

extends RigidBody3D
class_name DroppableBase

@export var stillness_threshold: int = 10
@export var time_until_static: float = 4.0

    ground_ray.enabled = true

    if is_static:
        return

    # Update raycast position and rotation every frame
    ground_ray.global_transform = Transform3D(
        Basis(),  # Reset rotation to identity
        global_transform.origin  # Maintain current global position

    enable_new_spawn()

    spawn_new_enabled.emit()
    queue_free()

Key improvements:

  1. Rotation-Independent Raycast: The raycast's transform is updated every frame to:
    • Match the RigidBody's current position (global_transform.origin)
    • Maintain a neutral rotation (Basis() = no rotation)
  2. Persistent Downward Direction: The target_position remains (0, -length, 0) in local space, but since we reset rotation every frame, this always points straight down in global space
  3. Accurate Ground Detection: Works even when the RigidBody is spinning or tumbling through the air

The raycast will now reliably detect ground collisions regardless of the RigidBody's rotation. The object will only become static when it's both motionless for the required time and has ground beneath it.

is it possible to check the height of the rigidbody3d to use as the ray length

Yes! We can calculate the object's height from its collision shapes and use that to dynamically set the ray length. Here's the improved version:

gdscript
extends RigidBody3D
class_name DroppableBase

@export var stillness_threshold: int = 10
@export var time_until_static: float = 4.0
@export var height_buffer: float = 0.1  # Small safety margin

var time_since_moved: float = 0.0
var is_static: bool = false
var ground_ray: RayCast3D

signal spawn_new_enabled

func _ready():
    ground_ray = RayCast3D.new()
    add_child(ground_ray)
    ground_ray.enabled = true

    # Calculate ray length based on object height
    var object_height = get_object_height()
    ground_ray.target_position = Vector3(0, -object_height, 0)

func get_object_height() -> float:
    var max_height: float = 0.0
    var min_height: float = 0.0

    # Check all collision shapes to find vertical bounds
    for child in get_children():
        if child is CollisionShape3D:
            var shape = child.shape
            var global_pos = child.global_transform.origin

            if shape is BoxShape3D:
                var extents = shape.size.y / 2
                max_height = max(max_height, global_pos.y + extents)
                min_height = min(min_height, global_pos.y - extents)
            elif shape is SphereShape3D:
                var radius = shape.radius
                max_height = max(max_height, global_pos.y + radius)
                min_height = min(min_height, global_pos.y - radius)
            elif shape is CapsuleShape3D:
                var height = shape.height
                var radius = shape.radius
                max_height = max(max_height, global_pos.y + height/2 + radius)
                min_height = min(min_height, global_pos.y - height/2 - radius)

    # Calculate total height with buffer
    return (max_height - min_height) + height_buffer

func _physics_process(delta: float) -> void:
    if is_static:
        return

    # Update ray position (but keep pointing straight down)
    ground_ray.global_transform = Transform3D(
        Basis(),
        global_transform.origin
    )

    var is_moving: bool = linear_velocity.length() > stillness_threshold

    if is_moving:
        time_since_moved = 0.0
    else:
        time_since_moved += delta
        if time_since_moved >= time_until_static and ground_ray.is_colliding():
            make_static()

    if global_transform.origin.y < -10:
        enable_new_spawn()

# Rest of the code remains the same as previous version

Key improvements:

  1. Dynamic Height Calculation:

    • Automatically calculates height from collision shapes
    • Works with boxes, spheres, and capsules
    • Adds a small buffer to ensure ground contact

  2. Adaptive Ray Length:

    • Ray length adjusts to the object's actual height
    • No more manual tuning of ground_check_ray_length

  3. Rotation-Proof:

    • Ray always points straight down in world space
    • Properly accounts for object rotation and spinning

The raycast will now always reach just past the object's lowest point, regardless of its rotation or shape. The object will only become static when:

  1. It's been motionless for the required time
  2. The dynamically calculated ray detects ground contact
  3. It's not falling through the world (y < -10 check)

Note: This assumes your collision shapes are properly set up and positioned relative to the RigidBody3D's origin. For complex shapes, you might need to add additional shape type checks.
